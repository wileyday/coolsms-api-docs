  "POST",
  "/SimpleMessage/3/getScheduledMessages",

      "offset": 0,
      "limit": 20
    }
  ),
  {"Authorization":"HMAC-SHA256 ApiKey=[API_KEY], Date=[DATE], Salt=[SALT], Signature=[SIGNATURE]"})
